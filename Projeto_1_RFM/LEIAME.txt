# Projeto 1 - Análise RFM (Recency, Frequency, Monetary Value) e Dashboard Interativo Para a Área de Marketing

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsabusp1 python=3.13

# Ative o ambiente:

conda activate dsabusp1 (ou: source activate dsabusp1)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute o jupyter notebook

jupyter notebook

# Acesse o jupyter pelo seu navegador padrão.

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsabusp1 --all
